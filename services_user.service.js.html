services/user.service.js

/**
 * This section is related to user services such as receiving profile information or changes in the profile section
 * @namespace Services.user_services
 */

import axios from "axios";
import authHeader from "./auth-header";
const qs = require("querystring");

const API_URL = "http://localhost:8000/";
/**
 *get data from '/profile/' with axios(with auth-header()) and retrun promise
 *@property {function} getProfile get user profile data service
 *@memberof Services.user_services
 * @returns {promises} - promises response of user profile data
 */
const getProfile = () => {
  return axios.get(API_URL + "profile/", { headers: authHeader() });
};
/**
 *put data to '/userinfo/' with axios(with auth-header()) and retrun promise
 *@property {function} editProfileUserInfo edit user profile data service
  *@param {Object} data object of user data({
    email: "*",
    first_name: "*",
    last_name: "*",
    city: "*",
    birthday: "*",
    ...
  })
 *@memberof Services.user_services
 * @returns {promises} - promises response of edit profile data
 */
const editProfileUserInfo = (data) => {
  return axios.put(API_URL + "userinfo/", qs.stringify(data), {
    headers: authHeader(),
  });
};
/**
 *send bloodTest And healthy information data with put action and '/profile/' url with axios(with auth-header()) and retrun promise
 *@property {function} editProfileProfileInfo edit user profile bloodTest and healthy information data service
 *@param {Object} data object of user bloodTest And healthy information
 *@memberof Services.user_services
 * @returns {promises} - promises response of edit profile data
 */
const editProfileProfileInfo = (data) => {
  return axios.put(API_URL + "profile/", qs.stringify(data), {
    headers: authHeader(),
  });
};
/**
 * retrun object for all auth services such as getProfile,editProfileUserInfo,editProfileProfileInfo
 *  @module userservices
 */
export default {
  getProfile,
  editProfileUserInfo,
  editProfileProfileInfo,
};
